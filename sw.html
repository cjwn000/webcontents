C언어 목차별 문제 풀이 학습

1. 변수와 자료형

문제 1: 두 정수의 합을 구하는 프로그램을 작성하세요.
#include <stdio.h>
int main() {
    int a, b;
    scanf("%d %d", &a, &b);
    printf("%d", a + b);
    return 0;
}
문제 2: 실수형 변수에 값을 저장하고 출력하세요.
#include <stdio.h>
int main() {
    float f = 3.14f;
    printf("%.2f", f);
    return 0;
}
문제 3: 문자형 변수에 문자 'A'를 저장하고 출력하세요.
#include <stdio.h>
int main() {
    char c = 'A';
    printf("%c", c);
    return 0;
}
문제 4: 정수형 변수 3개를 선언하고 각각 값을 입력받아 출력하세요.
#include <stdio.h>
int main() {
    int x, y, z;
    scanf("%d %d %d", &x, &y, &z);
    printf("%d %d %d", x, y, z);
    return 0;
}
문제 5: 변수의 값을 서로 바꾸는 프로그램을 작성하세요.
#include <stdio.h>
int main() {
    int a = 5, b = 10, temp;
    temp = a;
    a = b;
    b = temp;
    printf("%d %d", a, b);
    return 0;
}

2. 조건문

문제 1: 입력받은 수가 짝수인지 홀수인지 판별하세요.
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    if (n % 2 == 0) printf("짝수");
    else printf("홀수");
    return 0;
}
문제 2: 세 수 중 가장 큰 수를 출력하세요.
#include <stdio.h>
int main() {
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);
    int max = a;
    if (b > max) max = b;
    if (c > max) max = c;
    printf("%d", max);
    return 0;
}
문제 3: 점수를 입력받아 학점을 출력하세요.
#include <stdio.h>
int main() {
    int score;
    scanf("%d", &score);
    if (score >= 90) printf("A");
    else if (score >= 80) printf("B");
    else if (score >= 70) printf("C");
    else if (score >= 60) printf("D");
    else printf("F");
    return 0;
}
문제 4: 윤년 판별 프로그램을 작성하세요.
#include <stdio.h>
int main() {
    int year;
    scanf("%d", &year);
    if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0)
        printf("윤년");
    else
        printf("평년");
    return 0;
}
문제 5: switch문을 이용해 요일을 출력하세요.
#include <stdio.h>
int main() {
    int day;
    scanf("%d", &day);
    switch(day) {
        case 1: printf("월요일"); break;
        case 2: printf("화요일"); break;
        case 3: printf("수요일"); break;
        case 4: printf("목요일"); break;
        case 5: printf("금요일"); break;
        case 6: printf("토요일"); break;
        case 7: printf("일요일"); break;
        default: printf("잘못된 입력");
    }
    return 0;
}

3. 반복문

문제 1: 1부터 10까지 합을 구하세요.
#include <stdio.h>
int main() {
    int sum = 0;
    for (int i = 1; i <= 10; i++) sum += i;
    printf("%d", sum);
    return 0;
}
문제 2: 구구단 2단을 출력하세요.
#include <stdio.h>
int main() {
    for (int i = 1; i <= 9; i++)
        printf("2 x %d = %d\n", i, 2 * i);
    return 0;
}
문제 3: while문으로 1부터 5까지 출력하세요.
#include <stdio.h>
int main() {
    int i = 1;
    while (i <= 5) {
        printf("%d ", i);
        i++;
    }
    return 0;
}
문제 4: do-while문으로 10부터 1까지 출력하세요.
#include <stdio.h>
int main() {
    int i = 10;
    do {
        printf("%d ", i);
        i--;
    } while (i >= 1);
    return 0;
}
문제 5: break문을 이용해 1부터 100까지의 수 중 50에서 멈추는 프로그램을 작성하세요.
#include <stdio.h>
int main() {
    for (int i = 1; i <= 100; i++) {
        if (i == 51) break;
        printf("%d ", i);
    }
    return 0;
}
